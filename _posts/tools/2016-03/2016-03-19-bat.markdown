python server:
{% highlight bat %}
@ECHO off
E:
cd www
start python -m SimpleHTTPServer
{% endhighlight %}


jekyll server:
{% highlight bat %}
@echo off
call d:\Ruby22-x64\bin\setrbvars.bat
E:
cd www
start jekyll serve --baseurl '/blog'
{% endhighlight %}
